pipeline {
    agent any
    stages {
        stage('Version') {
            steps {
                echo 'Checking Java Version...'
                bat 'java -version'
                bat 'javac -version'
            }
        }
        stage('Build') {
            steps {
                echo 'Building...'
                script {
                    if (fileExists('add.java')) {
                        bat 'javac add.java'
                    } else {
                        error 'add.java file is missing in the workspace!'
                    }
                }
            }
        }
        stage('Test') {
            steps {
                echo 'Testing...'
                script {
                    if (fileExists('add.class')) {
                        bat 'java add'
                    } else {
                        error 'add.class not found! Ensure the build stage compiled successfully.'
                    }
                }
            }
        }
    }
}
